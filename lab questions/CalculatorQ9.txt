package com.example.calculatorq9;

import android.os.Bundle;
import android.view.View;
import android.widget.Button;
import android.widget.EditText;
import android.widget.TextView;
import android.widget.Toast;

import androidx.activity.EdgeToEdge;
import androidx.appcompat.app.AppCompatActivity;

public class MainActivity extends AppCompatActivity {
    TextView res;
    String currentinput="";
    String operator="";
    Double num1=0.0;
    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        EdgeToEdge.enable(this);
        setContentView(R.layout.activity_main);
        res=findViewById(R.id.textView);
        updateDisplay();
    }
    private void updateDisplay(){
        if(res != null){
            res.setText(currentinput.isEmpty()?"0":currentinput);
        }
    }
    public void numberClick(View view){
        currentinput+=view.getTag().toString();
        updateDisplay();
    }
    public void operatorClick(View view){
        if(!currentinput.isEmpty()){
            num1=Double.parseDouble(currentinput);
            operator=view.getTag().toString();
            currentinput=operator;
            updateDisplay();
            currentinput="";
        }
    }
    public void equalClick(View view){
        if(!currentinput.isEmpty()){
            Double num2=Double.parseDouble(currentinput);
            Double answer=calcOperations(num1,num2,operator);
            currentinput=answer.toString();
            updateDisplay();
            operator="";
        }
    }
    public void clearClick(View view){
        currentinput="";
        operator="";
        num1=0.0;
        updateDisplay();
    }
    Double calcOperations(Double a,Double b,String op){
        switch (op){
            case("+"):
                return (double) (a+b);
            case("-"):
                return (double) (a-b);
            case("*"):
                return (double) (a*b);
            case("/"):
                if (b == 0) {
                    Toast.makeText(this, "Cannot divide by zero", Toast.LENGTH_SHORT).show();
                    return 0.0;
                }
                return (double) (a / b);
        }
        return 0.0;
    }
}

package com.example.calculatorq9;

import android.os.Bundle;
import android.view.View;
import android.widget.Button;
import android.widget.EditText;
import android.widget.Toast;

import androidx.activity.EdgeToEdge;
import androidx.appcompat.app.AppCompatActivity;

public class MainActivity extends AppCompatActivity {

    EditText txt;
    String oldNumber = "";
    String operator = "";
    boolean isNewOp = true;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        EdgeToEdge.enable(this);
        setContentView(R.layout.activity_main);

        txt = findViewById(R.id.editTextText4);
        txt.setShowSoftInputOnFocus(false); // hide keyboard
    }

    public void operatorEvent(View view) {
        Button btn = (Button) view;
        String btnText = btn.getText().toString();
        String number = txt.getText().toString();

        switch (btnText) {
            // --- Numbers and dot ---
            case "0": case "1": case "2": case "3": case "4":
            case "5": case "6": case "7": case "8": case "9": case ".":
                if (isNewOp)
                    txt.setText("");
                isNewOp = false;
                txt.append(btnText);
                break;

            // --- Clear ---
            case "C":
                txt.setText("0");
                oldNumber = "";
                operator = "";
                isNewOp = true;
                break;

            // --- Operators ---
            case "+": case "-": case "*": case "/":
                oldNumber = txt.getText().toString();
                operator = btnText;
                isNewOp = true;
                break;

            // --- Equals (if you want to add one later) ---
            default:
                Toast.makeText(this, "Unsupported button: " + btnText, Toast.LENGTH_SHORT).show();
        }
    }

    // Optional: you can add another button (=) in XML and link it here
    public void equalEvent(View view) {
        String newNumber = txt.getText().toString();
        double result = 0.0;

        try {
            double num1 = Double.parseDouble(oldNumber);
            double num2 = Double.parseDouble(newNumber);

            switch (operator) {
                case "+": result = num1 + num2; break;
                case "-": result = num1 - num2; break;
                case "*": result = num1 * num2; break;
                case "/":
                    if (num2 == 0) {
                        Toast.makeText(this, "Cannot divide by zero!", Toast.LENGTH_SHORT).show();
                        return;
                    }
                    result = num1 / num2;
                    break;
            }
            txt.setText(result + "");
        } catch (Exception e) {
            Toast.makeText(this, "Error", Toast.LENGTH_SHORT).show();
        }
    }
}
